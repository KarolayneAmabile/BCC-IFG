{
    float x1, y1, x2, y2, d;

    printf("Defina Px1: ");
    scanf("%f", &x1);

    printf("Defina Py1: ");
    scanf("%f", &y1);

    printf("Defina Px2: ");
    scanf("%f", &x2);

    printf("Defina Py2: ");
    scanf("%f", &y2);

    printf("Insira b: ");
    scanf("%i", &b);

    printf("Insira c: ");
    scanf("%i", &c);

    printf("Insira d: ");
    scanf("%i", &d);

    printf("Insira e: ");
    scanf("%i", &e);

    printf("Insira f: ");
    scanf("%i", &f);

    x = (c*e - b*f)/(a*e - b*d);
    y = (a*f - c*d)/(a*e - b*d);

    printf("X: %i Y: %i", x, y);

}
